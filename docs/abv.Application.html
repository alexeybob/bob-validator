abv.Application

A abv.Application represents and manages your Validation application.

if ('' !== data && null !== data) {
// validation will fail
}

let form = validationEngine.make({
  name: 'required',
  name: 'required',
  email: 'required|email',
  birthday: 'required',
  creditCard: 'required',
  ip: 'required',
  locale: 'required',
  country: 'required',
  language: 'required',
  homepage: 'required'
}, {
  name: 'Leo Lane',
  email: 'leo.lane38@example.com',
  birthday: '03.07.1977',
  creditCard: '4111111111111111',
  ip: '8.8.8.8',
  locale: 'cy_GB',
  country: 'US',
  language: 'en_gb',
  homepage: 'https://github.com/alexeybob/bob-validator'
});

if (false === form['name'].isValid()) {
  form['name'].messages().first();
}

// Run validation
app.validate()

Stringlang 

The language used by the application. Defaults to 'en' (List of ISO 639-1 codes).

// Set the language for the application
this.app.lang = 'en';
